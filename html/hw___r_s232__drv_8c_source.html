107 void hw_RS232_Set_Baudrate(uint32_t BaudRate)
108 {
109 
110  uint32_t tmpreg = 0x00, apbclock = 0x00;
111  uint32_t integerdivider = 0x00;
112  uint32_t fractionaldivider = 0x00;
113  RCC_ClocksTypeDef RCC_ClocksStatus;
114 
115  /*---------------------------- USART BRR Configuration -----------------------*/
116  /* Configure the USART Baud Rate */
117  RCC_GetClocksFreq(&RCC_ClocksStatus);
118 
119  if ((p_Usart_Port == USART1) || (p_Usart_Port == USART6))
120  {
121  apbclock = RCC_ClocksStatus.PCLK2_Frequency;
122  }
123  else
124  {
125  apbclock = RCC_ClocksStatus.PCLK1_Frequency;
126  }
127 
128  /* Determine the integer part */
129  if ((p_Usart_Port->CR1 & USART_CR1_OVER8) != 0)
130  {
131  /* Integer part computing in case Oversampling mode is 8 Samples */
132  integerdivider = ((25 * apbclock) / (2 * (BaudRate)));
133  }
134  else /* if ((USARTx->CR1 & USART_CR1_OVER8) == 0) */
135  {
136  /* Integer part computing in case Oversampling mode is 16 Samples */
137  integerdivider = ((25 * apbclock) / (4 * (BaudRate)));
138  }
139 
140  tmpreg = (integerdivider / 100) << 4;
141  /* Determine the fractional part */
142  fractionaldivider = integerdivider - (100 * (tmpreg >> 4));
143 
144  /* Implement the fractional part in the register */
145  if ((p_Usart_Port->CR1 & USART_CR1_OVER8) != 0)
146  {
147  tmpreg |= ((((fractionaldivider * 8) + 50) / 100)) & ((uint8_t)0x07);
148  }
149  else /* if ((USARTx->CR1 & USART_CR1_OVER8) == 0) */
150  {
151  tmpreg |= ((((fractionaldivider * 16) + 50) / 100)) & ((uint8_t)0x0F);
152  }
153 
154 }

200 void USART6_IRQHandler(void) // UART6 Interrupt handler
201 {
