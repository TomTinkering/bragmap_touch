182 uint8_t RS232_GetByte(void)
183 {
184  uint32_t nrPackets;
185  hw_RS232_INT_t RX_Int_State = HW_GET_RX_INTERRUPT_STATE();
186  uint8_t result = 0xFF;
187 
188  // stop the Rx'er and get packet count (no guarantee reading the
189  // packet count is an atomic operation..)
190  HW_SET_INTERRUPT_STATE(HW_RS232_RX_INTERRUPT,HW_INT_DISABLED);
191  nrPackets = RS232_RX_Buffer.packetCount;
192  HW_SET_INTERRUPT_STATE(HW_RS232_RX_INTERRUPT,RX_Int_State);
193 
194  if(nrPackets)
195  {
196  //get byte from buffer and increment tail position. Since tailpos

198  result = RS232_RX_Buffer.pBuffer[RS232_RX_Buffer.tailPos];
199  RS232_RX_Buffer.tailPos ++;

203  //must be disabled

205  RS232_RX_Buffer.packetCount --;
206  HW_SET_INTERRUPT_STATE(HW_RS232_RX_INTERRUPT,RX_Int_State);
207 
208  }

224 rs232_err_t RS232_PutByte(uint8_t data)
225 {
226 
227  uint32_t remaining;
228  rs232_err_t result = RS232_BUFFER_FULL;
229  hw_RS232_INT_t TX_Int_State = HW_GET_TX_INTERRUPT_STATE();
230 
231  //Find amount of free space in the buffer, this is non atomic and volatile
232  //so interrupts must be disabled
233  HW_SET_INTERRUPT_STATE(HW_RS232_TX_INTERRUPT,HW_INT_DISABLED);
234  remaining = RS232_TX_Buffer.size - RS232_TX_Buffer.packetCount;
235  HW_SET_INTERRUPT_STATE(HW_RS232_TX_INTERRUPT,TX_Int_State);
236 
237  //if there is room in the buffer
238  if(remaining)
239  {
240  //Store the data
241  RS232_TX_Buffer.pBuffer[RS232_TX_Buffer.headPos] = data;
242  //Update the head position, no need to disable interrupts
243  RS232_TX_Buffer.headPos++;
244  RS232_TX_Buffer.headPos &= RS232_TX_BUFF_MASK;
245 
246  //update packet count. This is a non atomic operation on a volatile
247  //variable, so interrupts must be disabled. Since data is added to the
248  //buffer, the TX interrupt can be enabled afterwards
249  HW_SET_INTERRUPT_STATE(HW_RS232_TX_INTERRUPT,HW_INT_DISABLED);
250  RS232_TX_Buffer.packetCount ++;
251  HW_SET_INTERRUPT_STATE(HW_RS232_TX_INTERRUPT,HW_INT_ENABLED);
252 
253  //update result
254  result = RS232_OK;
255  }
256 
257  return result;
258 }

273 rs232_err_t RS232_GetByteBlock(uint8_t* BlockBuffer, uint32_t Count)
274 {
275 
276  uint32_t nrPackets,i;
277  hw_RS232_INT_t RX_Int_State = HW_GET_RX_INTERRUPT_STATE();
278 
279  // stop the Rx'er and get packet count (no guarantee reading the
280  // packet count is an atomic operation..)
281  HW_SET_INTERRUPT_STATE(HW_RS232_RX_INTERRUPT,HW_INT_DISABLED);
282  nrPackets = RS232_RX_Buffer.packetCount;
283  HW_SET_INTERRUPT_STATE(HW_RS232_RX_INTERRUPT,RX_Int_State);
284 
285  //check for the posibility of over-reading the buffer
286  if(Count > nrPackets){
287  return RS232_NOT_ENOUGH_ELEMENTS;
288  }
289 
290  //for each byte in cnt, copy source to destination
291  for (i=0; i<Count; i++){
292  //copy data
293  BlockBuffer[i] = RS232_RX_Buffer.pBuffer[RS232_RX_Buffer.tailPos];
294  //update tail position, no need to disable interrupts
295  RS232_RX_Buffer.tailPos ++;
296  RS232_RX_Buffer.tailPos &= RS232_RX_BUFF_MASK;
297  //update destination pointer
298  }
299 
300  //update packet count
301  RX_Int_State = HW_GET_RX_INTERRUPT_STATE();
302  HW_SET_INTERRUPT_STATE(HW_RS232_RX_INTERRUPT,HW_INT_DISABLED);
303  RS232_RX_Buffer.packetCount -= Count;
304  HW_SET_INTERRUPT_STATE(HW_RS232_RX_INTERRUPT,RX_Int_State);
305 
306  //return result
307  return RS232_OK;

322 rs232_err_t RS232_PutByteBlock(uint8_t* DataBlock, uint32_t Count)
323 {
324 
325  uint32_t i, remaining;
326  hw_RS232_INT_t TX_Int_State = HW_GET_TX_INTERRUPT_STATE();
327 
328  //calculate space in buffer
329  HW_SET_INTERRUPT_STATE(HW_RS232_TX_INTERRUPT,HW_INT_DISABLED);
330  remaining = RS232_TX_Buffer.size - RS232_TX_Buffer.packetCount;
331  HW_SET_INTERRUPT_STATE(HW_RS232_TX_INTERRUPT,TX_Int_State);
332 
333  if(remaining < Count){
334  return RS232_NOT_ENOUGH_SPACE;
335  }
336 
337  //for each byte in the source, store in TX buffer
338  for(i=0;i<Count;i++)
339  {
340  //Store data in TX buffer
341  RS232_TX_Buffer.pBuffer[RS232_TX_Buffer.headPos] = DataBlock[i];
342  //Update head pointer. No need to disable interrupts
343  RS232_TX_Buffer.headPos ++;
344  RS232_TX_Buffer.headPos &= RS232_TX_BUFF_MASK;
345  }
346 
347  //update packet count, and enable interrupt, since data was inserted
348  HW_SET_INTERRUPT_STATE(HW_RS232_TX_INTERRUPT,HW_INT_DISABLED);
349  RS232_TX_Buffer.packetCount += Count;
350  HW_SET_INTERRUPT_STATE(HW_RS232_TX_INTERRUPT,HW_INT_ENABLED);
351 
352  return RS232_OK;
353 }

492 {
493

495  //be fully filled since that would cause the tail and head positions
496  //to overlap. This is seen as an empty buffer.
497  if(RS232_RX_Buffer.packetCount < RS232_RX_FIFO_BUFF_SIZE)

504  //Increment packet count
