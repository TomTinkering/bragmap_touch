300 static uint8_t CTS_NVM_GetByteCRC8(uint8_t crc, uint8_t data)
301 {
302  uint8_t i;
303  //divisor polynominal, HD2 for 2048+ bits
304  const uint8_t poly = 0xA6;
305 
306  //calculate CRC
307  crc = crc ^ data;
308  for (i = 0; i < 8; i++) //for each bit position
309  {
310  if (crc & 0x01) //if LSB = 1
311  { //shift and XOR with polynomial

317  }
318  }
